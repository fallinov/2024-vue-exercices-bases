Utliser la méthode filter() pour filtrer les élèves par groupe.

Exemple d'utilisation de la méthode filter()

const personnes = [
  {id: 23, nom: 'Jean'},
  {id: 77, nom: 'Pauline'},
  {id: 1, nom: 'Fred'}
];

// Récupère toutes les personnes sauf la 77 => ou supprime la personne 77
let result = personnes.filter(function(val) {
  return val.id !== 77
});
console.log(result);

// Même chose en écriture optimisée
result = personnes.filter((val) => val.id !== 7)
console.log(result);
